00117 void Timer::SchedDecom(Agent* m, int t) {
00118   if (t < time_) {
00119     throw ValueError("Cannot schedule decommission for t < [current-time]");
00120   }
00121 
00122   // It is possible that a single agent may be scheduled for decommissioning
00123   // multiple times. If this happens, we cannot just add it to the queue again
00124   // - the duplicate entries will result in a double delete attempt and
00125   // segfaults and otherwise bad things.  Remove previous decommissionings
00126   // before scheduling this new one.
00127   std::map<int, std::vector<Agent*> >::iterator it;
00128   bool done = false;
00129   for (it = decom_queue_.begin(); it != decom_queue_.end(); ++it) {
00130     int t = it->first;
00131     std::vector<Agent*> ags = it->second;
00132     for (int i = 0; i < ags.size(); i++) {
00133       if (ags[i] == m) {
00134         CLOG(LEV_WARN) << "scheduled over previous decommissioning of " << m->id();
00135         decom_queue_[t].erase(decom_queue_[t].begin()+i);
00136         done = true;
00137         break;
00138       }
00139     }
00140     if (done) {
00141       break;
00142     }
00143   }
00144 
00145   decom_queue_[t].push_back(m);
00146 }
