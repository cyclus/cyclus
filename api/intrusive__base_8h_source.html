00013 ///
00014 /// Allows subclasses to track their reference count via the onboard
00015 /// "counter_" member variable. To use a class as a boost::intrusive_ptr,
00016 /// it should inherit from IntrusiveBase with default access (NOT
00017 /// public). All destructors of subclasses (and subsub, etc.) should be
00018 /// virtual to ensure memory deallocation occurs properly for objects
00019 /// that have been up-casted:
00020 /// @code
00021 /// class Resource: IntrusiveBase<Resource> {
00022 /// ...
00023 /// virtual ~Resource();
00024 /// ...
00025 /// }
00026 ///
00027 /// int main(...) {
00028 /// boost::intrusive_ptr<Resource> Resource(new Resource());

00031 /// // use resource as if it were a regular pointer, e.g.
00032 /// double quantity = resource->quantity();

00034 /// // equals operator compares raw pointer values:
00035 ///
00036 /// // always true
00037 /// if (resource == resource) {}
00038 ///

00040 /// if (resource == resource2) {}
00041 ///
00042 /// // don't worry about deallocation - it will be automatic.
