00550 void OsiCbcSolverInterface::applyColCut( const OsiColCut & cc )
00551 {
00552   const double * lower = modelPtr_->solver()->getColLower();
00553   const double * upper = modelPtr_->solver()->getColUpper();
00554   const CoinPackedVector & lbs = cc.lbs();
00555   const CoinPackedVector & ubs = cc.ubs();
00556   int i;
00557 
00558   for ( i=0; i<lbs.getNumElements(); i++ ) {
00559     int iCol = lbs.getIndices()[i];
00560     double value = lbs.getElements()[i];
00561     if ( value > lower[iCol] )
00562       modelPtr_->solver()->setColLower(iCol, value);
00563   }
00564   for ( i=0; i<ubs.getNumElements(); i++ ) {
00565     int iCol = ubs.getIndices()[i];
00566     double value = ubs.getElements()[i];
00567     if ( value < upper[iCol] )
00568       modelPtr_->solver()->setColUpper(iCol, value);
00569   }
00570 }
