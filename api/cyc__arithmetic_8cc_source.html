00015   // http://en.wikipedia.org/wiki/Kahan_summation_algorithm
00016   double y, t;
00017   double sum = 0.0;
00018   // A running compensation for lost low-order bits.
00019   double c = 0.0;
00020   for (std::vector<double>::iterator i = sorted.begin(); i != sorted.end();
00021        ++i) {
00022     y = *i - c;
00023     // So far, so good: c is zero.
00024     t = sum + y;
00025     // Alas, sum is big, y small, so low-order digits of y are lost.
00026     c = (t - sum) - y;
00027     // (t - sum) recovers the high-order part of y; subtracting y recovers -(low part of y)
00028     sum = t;

00030     // Next time around, the lost low part will be added to y in a fresh attempt.
00031   }
00032   return sum;
00033 }
