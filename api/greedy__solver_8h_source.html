00022 /// @brief A comparison function for sorting a container of Arcs by the
00023 /// requester's (unode's) preference, in decensing order (i.e., most preferred
00024 /// Arc first). In the case of a tie, a lexicalgraphic ordering of node ids is
00025 /// used.
00026 inline bool ReqPrefComp(const Arc& l, const Arc& r) {
00027   int lu = l.unode()->agent_id;
00028   int lv = l.vnode()->agent_id;
00029   int ru = r.unode()->agent_id;
00030   int rv = r.vnode()->agent_id;
00031   double lpref = l.unode()->prefs[l];
00032   double rpref = r.unode()->prefs[r];
00033   return (lpref != rpref) ? (lpref > rpref) : (lu > ru || (lu == ru && lv > rv));
00034 }
