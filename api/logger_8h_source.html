00021 /// @def LOG(level, prefix)
00022 ///
00023 /// allows easy logging via the streaming operator similar to std::cout;
00024 /// this is the primary way to use the Logging functionality.
00025 ///
00026 /// @param level LogLevel category or type of log statement.
00027 ///
00028 /// @param prefix A std::string value that functions as a unique
00029 /// identifier for the module. Prefixes longer than 6 characters will be
00030 /// truncated.

00032 /// @warning do not place any state-changing expressions with this macro
